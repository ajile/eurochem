    Handler for 404 status response. This middleware occures when nobody gave an answer for the request.

    var err = new Error('Not Found');
    err.status = 404;
    next(err);
});

    Development error handler will print stacktrace.

    if (app.get('env') === 'development') {
    app.use(function(err, req, res, next) {
        res.status(err.status || 500);
        res.render('error', {
            message: err.message,
            error: err
        });
    });
}
